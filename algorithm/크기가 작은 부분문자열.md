 # 크기가 작은 부분문자열



 ## 내 풀이

 ~~~ javascript
function solution(t, p) {
    let arr = ""
    let count = 0;
    let left = 0;
    let right = p.length;
    
    for (i=0; i<=t.length-p.length; i++){
        arr = t.slice(left, right);
        left += 1;
        right += 1;
        if(Number(arr) <= Number(p)){
            count++
        }
    }
    return count
}
 ~~~

## 베스트 풀이

 ~~~javasciprt
function solution(t, p) {
    let count = 0;
    for(let i=0; i<=t.length-p.length; i++) {
        let value = t.slice(i, i+p.length);
        if(+p >= +value) count++;
    }
    return count;
}
 ~~~
